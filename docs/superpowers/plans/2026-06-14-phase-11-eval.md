```python
import json
from pathlib import Path
from anatomy_eval.golden import load_golden, parse_golden_row

def promote_cases(cases: list[dict], regression_path) -> int:
    p = Path(regression_path); p.parent.mkdir(parents=True, exist_ok=True)
    existing = {it.id for it in load_golden(p)} if p.exists() else set()
    to_add, seen = [], set()
    for c in cases:
        cid = c.get("id")
        if cid in existing or cid in seen:
            continue
        parse_golden_row(c, 0)   # 非法即 raise（記憶體內，無 tmp 檔）
        to_add.append(c); seen.add(cid)
    if to_add:
        with p.open("a", encoding="utf-8") as f:   # 原子附加
            for c in to_add:
                f.write(json.dumps(c, ensure_ascii=False) + "\n")
    return len(to_add)
```
