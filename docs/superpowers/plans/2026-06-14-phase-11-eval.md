```python
"""自訂 RAGAS 指標（§7.3 out_of_scope_correctness）——純 Python、無 LLM、決定性。"""
from __future__ import annotations
import typing as t
from dataclasses import dataclass, field
from ragas.dataset_schema import SingleTurnSample
from ragas.metrics.base import MetricType, SingleTurnMetric
from ragas.run_config import RunConfig

OOS_PHRASE = "教材中查無此項"

@dataclass
class OutOfScopeCorrectness(SingleTurnMetric):
    name: str = "out_of_scope_correctness"
    _required_columns: dict[MetricType, set[str]] = field(
        default_factory=lambda: {MetricType.SINGLE_TURN: {"response", "reference"}})
    oos_phrase: str = OOS_PHRASE

    def init(self, run_config: RunConfig) -> None:  # Metric.init 為抽象
        pass

    async def _single_turn_ascore(self, sample: SingleTurnSample, callbacks) -> float:
        said = self.oos_phrase in (sample.response or "")
        should = self.oos_phrase in (sample.reference or "")
        return float(said == should)
```

```python
"""offline RAGAS 假件：零網路、決定性。研究 §4。"""
import hashlib
import numpy as np
from langchain_core.outputs import Generation, LLMResult
from ragas.llms.base import BaseRagasLLM
from ragas.embeddings.base import BaseRagasEmbeddings

class FakeRagasLLM(BaseRagasLLM):
    def _canned(self, prompt) -> str:
        s = prompt.to_string()
        if "statements" in s and "verdict" in s:
            return '{"statements":[{"statement":"s","reason":"r","verdict":1}]}'
        if "statements" in s:
            return '{"statements":["s"]}'
        if "verdict" in s:
            return '{"reason":"ok","verdict":1}'
        if "question" in s.lower():
            return '{"question":"q","noncommittal":0}'
        return "{}"
    def generate_text(self, prompt, n=1, temperature=0.01, stop=None, callbacks=None):
        return LLMResult(generations=[[Generation(text=self._canned(prompt))] * n])
    async def agenerate_text(self, prompt, n=1, temperature=None, stop=None, callbacks=None):
        return LLMResult(generations=[[Generation(text=self._canned(prompt))] * n])
    def is_finished(self, response) -> bool:
        return True

class FakeRagasEmbeddings(BaseRagasEmbeddings):
    dim = 8
    def _v(self, s: str):
        h = hashlib.sha256(s.encode()).digest()
        return (np.frombuffer(h[: self.dim], np.uint8) / 255.0).tolist()
    def embed_query(self, text): return self._v(text)
    def embed_documents(self, texts): return [self._v(t) for t in texts]
    async def aembed_query(self, text): return self._v(text)
    async def aembed_documents(self, texts): return [self._v(t) for t in texts]
```

```python
"""RAGAS 0.4.3 runner（mock-first、注入式 llm/embeddings）。§7.3 / D-J。

llm/embeddings MUST 非 None（否則 evaluate() 會去打 OpenAi——研究 §4 警告）。
CI/測試注入 BaseRagasLLM/BaseRagasEmbeddings 假件；真實 run 注入 LangchainLLMWrapper(ChatOpenAI(eval_key))。
contexts 餵 Docling MD（D-J）。
"""
from __future__ import annotations
from dataclasses import dataclass
from ragas import EvaluationDataset, SingleTurnSample, evaluate
from ragas.metrics._answer_relevance import ResponseRelevancy
from ragas.metrics._context_precision import LLMContextPrecisionWithoutReference
from ragas.metrics._faithfulness import Faithfulness
from anatomy_eval.ragas_metrics import OutOfScopeCorrectness

@dataclass(frozen=True)
class EvalRow:
    query: str
    retrieved_contexts: list[str]
    answer: str
    reference: str
    is_oos: bool = False

def default_metrics():
    # context_recall(LLM) 需 ground-truth 答案——v1 黃金題只有 expected_pages/concepts，
    # 故 v1 用 without-reference precision + faithfulness + relevancy + 純 Python OOS；
    # context_recall 待教師補 reference 答案後啟用（或改 NonLLMContextRecall 走 reference_contexts）。
    return [
        LLMContextPrecisionWithoutReference(),
        Faithfulness(),
        ResponseRelevancy(),
        OutOfScopeCorrectness(),
    ]

def run_eval(rows: list[EvalRow], *, llm, embeddings, metrics=None) -> dict[str, float]:
    if llm is None or embeddings is None:
        raise ValueError("llm 與 embeddings MUST 注入（mock-first，禁 evaluate() 的 OpenAI fallback）")
    samples = [SingleTurnSample(user_input=r.query, retrieved_contexts=r.retrieved_contexts,
                                response=r.answer, reference=r.reference) for r in rows]
    ds = EvaluationDataset(samples=samples)
    result = evaluate(ds, metrics=metrics or default_metrics(), llm=llm, embeddings=embeddings,
                      raise_exceptions=True, show_progress=False)
    df = result.to_pandas()
    names = [m.name for m in (metrics or default_metrics())]
    return {n: float(df[n].mean()) for n in names if n in df.columns}
```

```python
"""RAGAS gate（§7.3）：report 各指標 ≥ 門檻才過；否則 exit 1 並列出缺口。"""
from __future__ import annotations
import argparse, json, sys
from dataclasses import dataclass
from pathlib import Path
import yaml

@dataclass(frozen=True)
class GateResult:
    passed: bool
    failures: list[tuple[str, float, float]]  # (metric, actual, threshold)

def check_ragas(report: dict, thresholds: dict) -> GateResult:
    failures = []
    for metric, thr in thresholds.items():
        actual = report.get(metric)
        if actual is None or actual < thr:
            failures.append((metric, actual if actual is not None else float("nan"), thr))
    return GateResult(passed=not failures, failures=failures)

def main(argv=None) -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--report", required=True)
    ap.add_argument("--thresholds", required=True)
    args = ap.parse_args(argv)
    report = json.loads(Path(args.report).read_text(encoding="utf-8"))
    thr = yaml.safe_load(Path(args.thresholds).read_text(encoding="utf-8"))["ragas"]
    res = check_ragas(report, thr)
    if res.passed:
        print("RAGAS gate PASS"); return 0
    for m, a, t in res.failures:
        print(f"FAIL {m}: {a} < {t}")
    return 1

if __name__ == "__main__":
    sys.exit(main())
```

```python
"""recall@K by class gate（DL-013）。包 harness.evaluate_recall_by_class 的輸出做門檻比對。"""
from __future__ import annotations
from dataclasses import dataclass, field

@dataclass(frozen=True)
class RecallGateResult:
    passed: bool
    failures: list[tuple[str, float, float]] = field(default_factory=list)
    low_sample_warnings: list[str] = field(default_factory=list)

def check_recall(report: dict, thresholds: dict) -> RecallGateResult:
    by_class = report.get("by_class", {})
    n_by_class = report.get("n_by_class", {})
    min_n = thresholds.get("min_samples_per_class", 0)
    failures, warns = [], []
    for cls, thr in thresholds.get("by_class", {}).items():
        actual = by_class.get(cls)
        if actual is None or actual < thr:
            failures.append((cls, actual if actual is not None else float("nan"), thr))
        if n_by_class.get(cls, 0) < min_n:
            warns.append(cls)
    return RecallGateResult(passed=not failures, failures=failures, low_sample_warnings=warns)
```

```python
"""評估報告保留（§7.6：MUST 保留 ≥90 天）。prune 只刪 > retain_days 者，≤90d 一律保留。"""
from __future__ import annotations
import json
from datetime import datetime, timedelta
from pathlib import Path

def save_report(report: dict, directory, *, ts: datetime) -> Path:
    d = Path(directory); d.mkdir(parents=True, exist_ok=True)
    p = d / f"eval_report_{ts.strftime('%Y-%m-%dT%H-%M-%S')}.json"
    p.write_text(json.dumps(report, ensure_ascii=False), encoding="utf-8")
    return p

def list_reports(directory) -> list[Path]:
    return sorted(Path(directory).glob("eval_report_*.json"))

def prune_old(directory, *, now: datetime, retain_days: int = 90) -> int:
    if retain_days < 90:
        raise ValueError("retain_days 不得 < 90（§7.6 MUST 保留 ≥90 天）")
    cutoff = now - timedelta(days=retain_days)
    n = 0
    for p in list_reports(directory):
        stamp = p.stem.removeprefix("eval_report_")
        ts = datetime.fromisoformat(stamp.replace("-", ":", 2).replace("T", "T")) if False else \
             datetime.strptime(stamp, "%Y-%m-%dT%H-%M-%S").replace(tzinfo=now.tzinfo)
        if ts < cutoff:
            p.unlink(); n += 1
    return n
```

```python
"""Cohen's kappa（§7.6：多標註者 < 0.7 表題目不清需重寫）。手刻、無 sklearn。"""
from __future__ import annotations
from collections import Counter

def cohens_kappa(a: list[str], b: list[str]) -> float:
    if len(a) != len(b):
        raise ValueError("兩標註者長度須相同")
    if not a:
        raise ValueError("不可為空")
    n = len(a)
    po = sum(1 for x, y in zip(a, b) if x == y) / n
    ca, cb = Counter(a), Counter(b)
    labels = set(a) | set(b)
    pe = sum((ca.get(l, 0) / n) * (cb.get(l, 0) / n) for l in labels)
    return 1.0 if pe == 1.0 else (po - pe) / (1 - pe)

def kappa_gate(kappa: float, threshold: float = 0.7) -> bool:
    """True＝達標；False＝< 0.7（題目不清需重寫，§7.6）。"""
    return kappa >= threshold
```

```python
"""錯誤/👎 案例 → 回歸題庫促進（§7.4：錯誤案例 MUST 入回歸集，隔週須過）。

輸入 cases 為已由助教補上正解 expected 欄位的 golden 列（schema 同 §7.2）。
本函式只負責：schema 驗證（複用 golden 守門邏輯）+ 以 id 去重 + 附加。
"""
from __future__ import annotations
import json
from pathlib import Path
from anatomy_eval.golden import load_golden  # 複用 schema 驗證

def promote_cases(cases: list[dict], regression_path) -> int:
    p = Path(regression_path)
    existing_ids = {it.id for it in load_golden(p)} if p.exists() else set()
    # 先把新案寫到暫存以複用 load_golden 的逐行 schema 守門
    to_add, new_ids = [], set()
    for c in cases:
        cid = c.get("id")
        if cid in existing_ids or cid in new_ids:
            continue
        # 驗證單列：寫暫存單行 → load_golden（會 raise 非法 schema / should_refuse）
        tmp = p.with_suffix(".tmp.jsonl")
        tmp.write_text(json.dumps(c, ensure_ascii=False) + "\n", encoding="utf-8")
        load_golden(tmp)  # 非法即 raise
        tmp.unlink()
        to_add.append(c); new_ids.add(cid)
    if to_add:
        with p.open("a", encoding="utf-8") as f:
            for c in to_add:
                f.write(json.dumps(c, ensure_ascii=False) + "\n")
    return len(to_add)
```

```python
"""抽檢資料層（§7.4）。Streamlit shell 只做 UI，邏輯在此（可測）。"""
from __future__ import annotations
import json, random
from pathlib import Path

ALLOWED_LABELS = ("correct", "partial", "wrong")

def sample_logs(rows: list[dict], n: int, *, seed: int) -> list[dict]:
    rng = random.Random(seed)
    return rng.sample(rows, min(n, len(rows)))

def export_annotations(annotations: list[dict], path) -> None:
    for a in annotations:
        if a.get("label") not in ALLOWED_LABELS:
            raise ValueError(f"label 須為 {ALLOWED_LABELS}，收到 {a.get('label')!r}")
    with Path(path).open("w", encoding="utf-8") as f:
        for a in annotations:
            f.write(json.dumps(a, ensure_ascii=False) + "\n")
```
